// ── INIT ─────────────────────────────────────────────────────
document.addEventListener('DOMContentLoaded', () => {

});

function cargarTodo() {
  Promise.all([
    new Promise(r => google.script.run.withSuccessHandler(r).getPipeline()),
    new Promise(r => google.script.run.withSuccessHandler(r).getActividades())
  ]).then(([p, a]) => {
    pipeline    = p || [];
    actividades = a || [];
    renderAll();
  });
}

function renderAll() {
  poblarFiltros();
  filtrarPipeline();
  renderPendientes();
  filtrarActividades();
  renderStats();
  poblarSelectEmpresas();
}

// ── TABS ─────────────────────────────────────────────────────
function switchTab(tab, el) {
  document.querySelectorAll('.view').forEach(v => v.classList.remove('active'));
  document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
  document.getElementById('view-' + tab).classList.add('active');
  el.classList.add('active');
}

function poblarFiltros() {
  const rubros    = [...new Set(pipeline.map(e => e.rubro).filter(Boolean))].sort();
  const segmentos = [...new Set(pipeline.map(e => e.segmento).filter(Boolean))].sort();
  const resps     = [...new Set(actividades.map(a => a.responsable).filter(Boolean))].sort();

  poblarSelect('fil-rubro',       rubros,    'Todos los rubros');
  poblarSelect('fil-segmento',    segmentos, 'Todos los segmentos');
  poblarSelect('fil-responsable', resps,     'Todos');
}

function poblarSelect(id, items, emptyLabel) {
  const sel = document.getElementById(id);
  const val = sel.value;
  sel.innerHTML = `<option value="">${emptyLabel}</option>`;
  items.forEach(i => {
    const o = document.createElement('option');
    o.value = o.textContent = i;
    if (i === val) o.selected = true;
    sel.appendChild(o);
  });
}

function poblarSelectEmpresas() {
  const sel = document.getElementById('na-empresa');
  const sorted = [...pipeline].sort((a,b) => a.empresa.localeCompare(b.empresa));
  sel.innerHTML = '<option value="">— Seleccionar empresa —</option>';
  sorted.forEach(e => {
    const o = document.createElement('option');
    o.value = e.id;
    o.textContent = `${e.empresa} (${e.id})`;
    o.dataset.empresa = e.empresa;
    sel.appendChild(o);
  });
}

function filtrarPipeline() {
  const rubro    = document.getElementById('fil-rubro').value.toLowerCase();
  const etapa    = document.getElementById('fil-etapa').value.toLowerCase();
  const segmento = document.getElementById('fil-segmento').value.toLowerCase();
  const buscar   = document.getElementById('fil-buscar').value.toLowerCase();

  pipelineFiltrado = pipeline.filter(e => {
    if (rubro    && !e.rubro.toLowerCase().includes(rubro))       return false;
    if (etapa    && e.etapa.toLowerCase() !== etapa)              return false;
    if (segmento && e.segmento.toLowerCase() !== segmento)        return false;
    if (buscar   && !(
      e.empresa.toLowerCase().includes(buscar) ||
      e.contacto.toLowerCase().includes(buscar) ||
      e.mail.toLowerCase().includes(buscar)
    )) return false;
    return true;
  });

  renderTablaPipeline();
  renderStats();
  document.getElementById('cnt-pipeline').textContent = pipelineFiltrado.length;
}

function limpiarFiltros() {
  ['fil-rubro','fil-etapa','fil-segmento'].forEach(id => document.getElementById(id).value = '');
  document.getElementById('fil-buscar').value = '';
  filtrarPipeline();
}

// ── PENDIENTES ────────────────────────────────────────────────
function renderPendientes() {
  const hoy = new Date(); hoy.setHours(0,0,0,0);

  const pendientes = pipeline
    .filter(e => e.fechaProx)
    .map(e => ({...e, _fecha: new Date(e.fechaProx)}))
    .sort((a,b) => a._fecha - b._fecha);

  const vencidas  = pendientes.filter(e => e._fecha < hoy);
  const deHoy     = pendientes.filter(e => e._fecha.getTime() === hoy.getTime());
  const proximas  = pendientes.filter(e => e._fecha > hoy);

  document.getElementById('cnt-pendientes').textContent = pendientes.length;

  let html = '';

  if (vencidas.length) {
    html += `<div class="pending-section">
      <div class="pending-title">🔴 Vencidas (${vencidas.length})</div>
      ${vencidas.map(e => cardPendiente(e, 'vencida')).join('')}
    </div>`;
  }
  if (deHoy.length) {
    html += `<div class="pending-section">
      <div class="pending-title" style="color:var(--accent4)">🟢 Hoy (${deHoy.length})</div>
      ${deHoy.map(e => cardPendiente(e, 'hoy')).join('')}
    </div>`;
  }
  if (proximas.length) {
    html += `<div class="pending-section">
      <div class="pending-title" style="color:var(--accent2)">🔵 Próximas (${proximas.length})</div>
      ${proximas.map(e => cardPendiente(e, '')).join('')}
    </div>`;
  }

  if (!html) html = '<div class="empty">No hay acciones con fecha asignada.</div>';
  document.getElementById('contenido-pendientes').innerHTML = html;
}

function cardPendiente(e, cls) {
  return `
    <div class="pending-card ${cls}" onclick="abrirDetalle('${e.id}')">
      <div class="pending-date">${e.fechaProx}</div>
      <div class="pending-info">
        <div class="pending-empresa">${e.empresa}</div>
        <div class="pending-accion">${e.proxAccion}</div>
      </div>
      ${badgeEtapa(e.etapa)}
    </div>
  `;
}

// ── PANEL DETALLE ─────────────────────────────────────────────
function abrirDetalle(id) {
  const e = pipeline.find(x => x.id === id);
  if (!e) return;
  detalleEmpresaId = id;

  document.getElementById('dp-id').textContent      = e.id;
  document.getElementById('dp-empresa').textContent  = e.empresa;
  document.getElementById('dp-badge').innerHTML      = badgeEtapa(e.etapa);

  const info = [
    ['Segmento',    e.segmento],
    ['Rubro',       e.rubro],
    ['Contacto',    e.contacto],
    ['Cargo',       e.cargo],
    ['Mail',        e.mail],
    ['Teléfono',    e.tel],
    ['CEO / Ref.',  e.ceo],
    ['Próx. Acción',e.proxAccion],
    ['Fecha',       e.fechaProx],
  ].filter(([,v]) => v).map(([k,v]) => `
    <div class="detail-row">
      <div class="detail-key">${k}</div>
      <div class="detail-val">${v}</div>
    </div>
  `).join('');

  document.getElementById('dp-info').innerHTML = info;

  const acts = actividades
    .filter(a => a.idCliente === id)
    .sort((a,b) => b.fecha.localeCompare(a.fecha));

  document.getElementById('dp-actividades').innerHTML = acts.length
    ? acts.map(a => `
        <div class="act-item">
          <div class="act-item-top">
            <span>${a.fecha}</span>
            <span>${a.canal} · ${a.responsable}</span>
          </div>
          <div class="act-item-detalle">${a.detalle}</div>
          ${a.resultado ? `<div class="act-item-result">→ ${a.resultado}</div>` : ''}
          ${a.proxPaso  ? `<div style="color:var(--muted);font-size:10px;margin-top:2px">Próx: ${a.proxPaso}${a.fechaProx ? ' · ' + a.fechaProx : ''}</div>` : ''}
        </div>
      `).join('')
    : '<div style="color:var(--muted);font-size:11px">Sin actividades registradas.</div>';

  document.getElementById('detail-panel').classList.add('open');
}

function cerrarDetalle() {
  document.getElementById('detail-panel').classList.remove('open');
  detalleEmpresaId = null;
}

function abrirActividadDesdeDetalle() {
  cerrarDetalle();
  const e = pipeline.find(x => x.id === detalleEmpresaId || x.id === document.getElementById('dp-id').textContent);
  abrirModalActividad(detalleEmpresaId || (e && e.id));
}

function guardarEmpresa() {
  const datos = {
    empresa:    v('ne-empresa'),
    segmento:   v('ne-segmento'),
    rubro:      v('ne-rubro'),
    contacto:   v('ne-contacto'),
    cargo:      v('ne-cargo'),
    mail:       v('ne-mail'),
    tel:        v('ne-tel'),
    ceo:        v('ne-ceo'),
    etapa:      v('ne-etapa'),
    proxAccion: v('ne-proxAccion'),
    fechaProx:  v('ne-fechaProx'),
  };
  if (!datos.empresa) { toast('Ingresá el nombre de la empresa', true); return; }

  google.script.run
    .withSuccessHandler(res => {
      cerrarModal('modal-empresa');
      toast(`✓ Empresa ${res.id} agregada al pipeline`);
      cargarTodo();
      resetForm('modal-empresa');
    })
    .withFailureHandler(err => toast('Error: ' + err.message, true))
    .agregarEmpresa(datos);
}

// ── GUARDAR ACTIVIDAD ─────────────────────────────────────────
function guardarActividad() {
  const sel = document.getElementById('na-empresa');
  const idCliente = sel.value;
  const empresaNombre = sel.options[sel.selectedIndex]?.dataset?.empresa || '';

  if (!idCliente) { toast('Seleccioná una empresa', true); return; }
  if (!v('na-canal'))   { toast('Seleccioná un canal', true); return; }
  if (!v('na-detalle')) { toast('Completá el detalle', true); return; }

  const datos = {
    idCliente,
    empresa:     empresaNombre,
    fecha:       v('na-fecha'),
    canal:       v('na-canal'),
    responsable: v('na-responsable'),
    detalle:     v('na-detalle'),
    resultado:   v('na-resultado'),
    proxPaso:    v('na-proxPaso'),
    fechaProx:   v('na-fechaProx'),
  };

  google.script.run
    .withSuccessHandler(() => {
      cerrarModal('modal-actividad');
      toast('✓ Actividad registrada');
      cargarTodo();
      resetForm('modal-actividad');
    })
    .withFailureHandler(err => toast('Error: ' + err.message, true))

}

// ── HELPERS ───────────────────────────────────────────────────
function v(id) { return (document.getElementById(id)?.value || '').trim(); }

function resetForm(modalId) {
  document.getElementById(modalId).querySelectorAll('input, textarea, select').forEach(el => {
    if (el.tagName === 'SELECT') el.selectedIndex = 0;
    else el.value = '';
  });
}

function badgeEtapa(etapa) {
  const map = {
    'Prospecto':        'prospecto',
    'Contactado':       'contactado',
    'Propuesta Enviada':'propuesta',
    'Negociación':      'negociacion',
    'Cerrado':          'cerrado',
    'Descartado':       'descartado',
  };
  const cls = map[etapa] || 'prospecto';
  return `<span class="badge badge-${cls}">${etapa || '—'}</span>`;
}

function badgeCanal(canal) {
  const icons = { LinkedIn:'🔗', Mail:'📧', WhatsApp:'💬', Llamada:'📞', 'Página Web':'🌐', Presencial:'🤝', Otro:'📌' };
  return `<span style="font-size:11px">${icons[canal]||'📌'} ${canal}</span>`;
}

function badgeResultado(res) {
  let color = 'var(--muted)';
  if (res?.includes('positivo') || res?.includes('pactada')) color = 'var(--accent4)';
  else if (res?.includes('negativo') || res?.includes('rebotado')) color = 'var(--accent3)';
  else if (res?.includes('evaluación')) color = 'var(--accent2)';
  return `<span style="font-size:11px;color:${color}">${res||'—'}</span>`;
}

function fechaDisplay(f) {
  if (!f) return '—';
  const hoy = new Date(); hoy.setHours(0,0,0,0);
  const d = new Date(f);
  const diff = Math.round((d - hoy) / 86400000);
  if (diff < 0)  return `<span style="color:var(--accent3)">${f}</span>`;
  if (diff === 0)return `<span style="color:var(--accent4)">HOY</span>`;
  if (diff <= 3) return `<span style="color:var(--accent)">${f}</span>`;
  return f;
}

function toast(msg, error=false) {
  const t = document.getElementById('toast');
  t.textContent  = msg;
  t.style.background = error ? 'var(--accent3)' : 'var(--accent)';
  t.style.color      = error ? '#fff' : '#0d0f14';
  t.style.display = 'block';
  setTimeout(() => t.style.display = 'none', 3000);
}
